    class Func : public Factory {
     public:
        DynamicInput x{"x", 3};
        StaticOutput y{"y", 2};
    
        void generate() {
            y[0] = x[0] * x[1] + x[2];
            y[1] = diff(y[0], x[0]);
        }
    };
    OUTPUT_CXX_CODE_MAIN(Func, ns, cls);
    
    ns::cls f(y);
    f(x);
    std::cout << y[0] << ", " << y[1] << std::endl;
